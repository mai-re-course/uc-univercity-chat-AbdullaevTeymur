@startuml
actor Пользователь
actor Система
actor Бэкенд

Пользователь -> Система : начинает регистрацию, нажав на кнопку регистрация
Пользователь <-- Система : запрашивает способ регистрации (по номеру телефона , Email адресу)
alt по номеру телефона
	Пользователь -> Система :  выбирает способ регистрации по номеру телефона, нажав на кнопку и вводит номер телефона
	Система -> Бэкенд : отсылает введенный номер
	Пользователь <- Бэкенд : отсылает sms сообщение с регистрационным кодом на введенный номер
else по Email адресу
	Пользователь -> Система :  выбирает способ регистрации по Email адресу, нажав на кнопку и вводит Email адрес
	Система -> Бэкенд : отсылает введенный Email адрес
	Пользователь <- Бэкенд : отсылает сообщение с регистрационным кодом на введенный Email адрес
end
Пользователь <-- Система : запрашивает регистрационный код
Пользователь -> Система : вводит регистрационный код
Система -> Бэкенд : отсылает введенный код для проверки на сходимость
Система <-- Бэкенд : отсылает подтверждение
Пользователь <-- Система : запрашивает данные ( ФИО, Должность, учебную группу(для должности студент), дату рождения, пароль, подтверждение пароля, согласие на обработку персональных данных)
Пользователь -> Система : вводит запрашиваемые данные
Система -> Система : проверяет данные на соответствие требованиям шаблона
Система -> Бэкенд : отправляет данные 
Бэкенд -> Бэкенд : добавляет данные о пользователе в базу данных
Система <-- Бэкенд : возвращает ответ успешной регистрации
Система -> Система : выводит на экран запись об успешной регистрации
@enduml
